  • /* A typographic palette for code & design. */
  •  
  • font-size: 16vw;
  •  
  • /* set variation axis values */
  • --prop: "PROP" 0; Mono Sans /* Proportion: Mono (fixed-width) or Sans (natural-width) */
  • --xprn: "XPRN" 0; /* Expression: Linear to Casual */
  • --wght: "wght" 800; /* Weight: Light to Black – also set with `font-weight` */
  • --slnt: "slnt" 0; /* Slant: 0 to -15 degrees */
  • --ital: "ital" 0.5; Off Auto On /* Italic: always roman, auto, or always italic */
  •  
  • /* apply axis values with CSS custom properties */
  • font-variation-settings: var(--prop), var(--xprn), var(--wght), var(--slnt), var(--ital);
